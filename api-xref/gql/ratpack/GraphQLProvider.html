1   package gql.ratpack;
2   
3   import javax.inject.Provider;
4   import java.util.Optional;
5   
6   import com.google.inject.Inject;
7   import graphql.GraphQL;
8   import graphql.execution.instrumentation.Instrumentation;
9   import graphql.schema.GraphQLSchema;
10  
11  /**
12   * Creates an instance of {@link GraphQL} that can be reused among all
13   * incoming requests.
14   *
15   * @since 0.3.2
16   */
17  public class GraphQLProvider implements Provider<GraphQL> {
18  
19    private GraphQLSchema graphQLSchema;
20    private Instrumentation instrumentation;
21  
22    /**
23     * Sets the {@link GraphQLSchema} instance. This is a hard
24     * dependency
25     *
26     * @since 0.3.2
27     */
28    @Inject
29    public void setGraphQLSchema(GraphQLSchema schema) {
30      this.graphQLSchema = schema;
31    }
32  
33    /**
34     * Sets the {@link Instrumentation} instance used. This is a soft
35     * dependency
36     *
37     * @since 0.3.2
38     */
39    @Inject(optional = true)
40    public void setInstrumentation(Instrumentation instrumentation) {
41      this.instrumentation = instrumentation;
42    }
43  
44    @Override
45    public GraphQL get() {
46      final GraphQL.Builder builder = GraphQL.newGraphQL(graphQLSchema);
47  
48      return Optional
49        .ofNullable(instrumentation)
50        .map(ins -> builder.instrumentation(ins).build())
51        .orElse(builder.build());
52    }
53  }
