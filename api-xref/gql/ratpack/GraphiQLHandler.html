1   package gql.ratpack;
2   
3   import java.net.URISyntaxException;
4   import java.net.URL;
5   import java.nio.file.Path;
6   
7   import ratpack.handling.Context;
8   import ratpack.handling.Handler;
9   
10  /**
11   * GraphiQL endpoint. This handler will be exposing a given GraphiQL
12   * client.
13   *
14   * By default the handler is disable, so in order to enable it you
15   * should set the {@link GraphQLModuleConfig}#activateGraphiQL to
16   * true when enabling the module
17   *
18   * @since 0.2.0
19   */
20  public class GraphiQLHandler implements Handler {
21  
22    @Override
23    public void handle(Context ctx) throws Exception {
24      GraphQLModuleConfig config = ctx.get(GraphQLModuleConfig.class);
25  
26      if (config.activateGraphiQL) {
27        ctx.render(getGraphiQLFile());
28      } else {
29        ctx.getResponse().status(404);
30        ctx.getResponse().send();
31      }
32    }
33  
34    private Path getGraphiQLFile() throws URISyntaxException {
35      URL url = GraphiQLHandler.class.getResource("/static/index.html");
36  
37      return PathUtil.toPath(url);
38    }
39  }
