1   package gql.ratpack;
2   
3   import com.google.inject.Scopes;
4   import graphql.GraphQL;
5   import ratpack.guice.ConfigurableModule;
6   
7   /**
8    * Configures GraphQL and GraphiQL handler instances
9    *
10   * import static ratpack.groovy.Groovy.ratpack
11   *
12   * import gql.ratpack.GraphQLHandler
13   * import gql.ratpack.GraphiQLHandler
14   *
15   * ratpack {
16   *   bindings {
17   *     module(GraphQLModule)
18   *   }
19   *   handlers {
20   *     prefix('graphql') {
21   *       post(GraphQLHandler)
22   *       prefix('browser') {
23   *         get(GraphiQLHandler)
24   *       }
25   *     }
26   *   }
27   * }
28   *
29   * @since 0.2.0
30   */
31  public class GraphQLModule extends ConfigurableModule<GraphQLModuleConfig> {
32  
33    @Override
34    public void configure() {
35      bind(GraphQL.class)
36        .toProvider(GraphQLProvider.class)
37        .in(Scopes.SINGLETON);
38  
39      bind(GraphQLHandler.class).in(Scopes.SINGLETON);
40      bind(gql.ratpack.pac4j.GraphQLHandler.class).in(Scopes.SINGLETON);
41      bind(GraphiQLHandler.class).in(Scopes.SINGLETON);
42    }
43  }
