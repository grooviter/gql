1   package gql.ratpack;
2   
3   import static ratpack.util.Exceptions.uncheck;
4   
5   import java.io.IOException;
6   import java.net.URI;
7   import java.net.URISyntaxException;
8   import java.net.URL;
9   import java.nio.file.FileSystem;
10  import java.nio.file.FileSystemNotFoundException;
11  import java.nio.file.FileSystems;
12  import java.nio.file.Path;
13  import java.nio.file.Paths;
14  
15  import com.google.common.collect.ImmutableMap;
16  
17  /**
18   * Helps to resolve resources location
19   *
20   * @since 0.2.0
21   */
22  public final class PathUtil {
23  
24    /**
25     * Resolves a given resource location whether it's located in the
26     * current classpath or in any of the available jars in the
27     * application
28     *
29     * @param resource the {@link URL} of the resource
30     * @return a full {@link Path} where the resource can be located
31     * @since 0.2.0
32     */
33    public static Path toPath(URL resource) throws URISyntaxException {
34      URI uri = resource.toURI();
35  
36      String scheme = uri.getScheme();
37      if (scheme.equals("file")) {
38        return Paths.get(uri);
39      }
40  
41      if (!scheme.equals("jar")) {
42        throw new IllegalStateException("Cannot deal with class path resource url: " + uri);
43      }
44  
45      String s = uri.toString();
46      int separator = s.indexOf("!/");
47      String entryName = s.substring(separator + 2);
48      URI fileURI = URI.create(s.substring(0, separator));
49      FileSystem fs = null;
50      try {
51        // Check if there's an existing file system, since it's provider-dependent whether file systems with the same URI are allowed
52        fs = FileSystems.getFileSystem(fileURI);
53        if (!fs.isOpen()) { // It's provider-dependent whether to return closed file systems
54                           fs = null; // Ignore it; closed file systems can't be used
55        }
56      } catch (FileSystemNotFoundException ignore) {
57        // Continue to create the file system
58      }
59      if (fs == null) {
60        try {
61          fs = FileSystems.newFileSystem(fileURI, ImmutableMap.of());
62        } catch (IOException e) {
63          throw uncheck(e);
64        }
65      }
66      return fs.getPath(entryName);
67    }
68  }
