1   package gql.ratpack.exec;
2   
3   import java.util.concurrent.CompletableFuture;
4   import java.util.function.Supplier;
5   
6   import ratpack.exec.Execution;
7   
8   /**
9    * Creates instances of {@link CompletableFuture} honoring Ratpack's executors
10   *
11   * @since 0.1.0
12   */
13  public class Futures {
14  
15    /**
16     * Creates a {@link CompletableFuture} using the blocking executor
17     *
18     * @param supplier code providing a value
19     * @return a blocking instance of {@link CompletableFuture}
20     * @since 0.1.0
21     */
22    public static <T> CompletableFuture<T> blocking(Supplier<T> supplier) {
23      return CompletableFuture.supplyAsync(supplier, Execution.current().getController().getBlockingExecutor());
24    }
25  
26    /**
27     * Creates a {@link CompletableFuture} using the non blocking executor (event loop)
28     *
29     * @param supplier code providing a value
30     * @return a non blocking instance of {@link CompletableFuture}
31     * @since 0.1.0
32     */
33    public static <T> CompletableFuture<T> async(Supplier<T> supplier) {
34      return CompletableFuture.supplyAsync(supplier, Execution.current().getController().getExecutor());
35    }
36  }
