1   package gql.ratpack.pac4j;
2   
3   import static gql.ratpack.GraphQLHandlerUtil.executeGraphQL;
4   import static gql.ratpack.GraphQLHandlerUtil.renderGraphQL;
5   import static gql.ratpack.GraphQLHandlerUtil.renderGraphQLError;
6   
7   import java.util.Map;
8   import java.util.Optional;
9   
10  import com.fasterxml.jackson.core.JsonParseException;
11  import org.pac4j.core.profile.UserProfile;
12  import ratpack.handling.Context;
13  import ratpack.handling.Handler;
14  import ratpack.pac4j.RatpackPac4j;
15  
16  /**
17   * GraphQL endpoint making available Pac4j UserProfile in
18   * an easy way through Ratpack's {@link Context} request
19   *
20   * @since 0.3.1
21   */
22  public class GraphQLHandler implements Handler {
23  
24    @Override
25    public void handle(Context ctx) {
26      RatpackPac4j
27        .userProfile(ctx)
28        .flatMap((Optional<UserProfile> userProfile) -> {
29          // making UserProfile available easily through
30          // ctx.request.get(UserProfile)
31          ctx.getRequest().add(userProfile);
32  
33          return ctx
34            .parse(Map.class)
35            .onError(JsonParseException.class, renderGraphQLError(ctx))
36            .flatMap(executeGraphQL(ctx));
37        }).then(renderGraphQL(ctx));
38    }
39  }
